## 2. Source module — `src/sources/<id>/index.js`

```js
/**
 * <Name> source
 *
 * Brief description of the site and what kind of listings it has.
 * Target URLs are read from <ID_UPPERCASE>_URLS (comma-separated) in .env.
 * loginRequired: true/false — cookies stored in storage/<id>-cookies.json
 */

const path = require('path');

const SOURCE_ID = '<id>';

function normalizeTargetUrl(url) {
  return url.split('?')[0].split('&')[0];
}

function getTargets({ urls = [], env = {} } = {}) {
  if (urls.length) return urls.filter(Boolean);

  const configuredUrls = env.<ID_UPPERCASE>_URLS;
  if (!configuredUrls) return [];

  return configuredUrls.split(',').map((url) => url.trim()).filter(Boolean);
}

async function extractListings(page) {
  return page.evaluate(() => {
    const results = [];

    document.querySelectorAll('LISTING_SELECTOR').forEach((item) => {
      const rawText = item.innerText?.trim() || '';
      if (rawText.length < 10) return;

      const linkEl = item.querySelector('a[href]');
      const url = linkEl ? linkEl.href.split('?')[0] : 'none';
      const imgEls = Array.from(item.querySelectorAll('img'))
        .filter(img => img.width > 80)
        .map(img => img.src);

      // Build a stable id. Prefer a unique id embedded in the URL.
      const idMatch = url.match(/\/(\d+)/);
      const rawId = idMatch ? idMatch[1] : rawText.replace(/[^a-zA-Z0-9]/g, '').slice(0, 30);

      const priceMatch = rawText.match(
        /(?:CHF|€|EUR|USD|\$)\s?([\d'.\s]{3,8})|([\d'.\s]{3,8})\s?(?:CHF|€|EUR)/i,
      );
      let price = null;
      if (priceMatch) {
        price = parseInt((priceMatch[1] || priceMatch[2]).replace(/[^\d]/g, ''), 10);
        if (Number.isNaN(price)) price = null;
      }

      // For structured sources, extract more fields here
      // const roomsMatch = rawText.match(...);

      results.push({ rawId, url, address_raw: rawText, image_urls: imgEls, price });
    });

    return results;
  }).then(raw => raw.map(item => ({
    id:             `${SOURCE_ID}_${item.rawId}`,
    source:         SOURCE_ID,
    url:            item.url,
    address_raw:    item.address_raw,
    image_urls:     item.image_urls ?? [],
    price:          item.price ?? null,
    currency:       'CHF',
    price_period:   'month',
    rooms:          null,           // fill in if parseable
    living_space_m2: null,          // fill in if parseable
    floor:          null,
    total_floors:   null,
    street:         null,
    street_number:  null,
    zip_code:       null,           // fill in if parseable
    city:           null,           // fill in if parseable
    country_code:   'CH',
    latitude:       null,
    longitude:      null,
    title:          null,           // fill in if available
    description:    null,           // fill in if available
    listing_type:   'rent',
    property_type:  null,
    available_from: null,
  })));
}

module.exports = {
  id: SOURCE_ID,
  name: '<Name>',
  loginRequired: false,
  loginUrl: null,
  maxScrolls: 40,
  initialDelayMs: 3000,
  scrollDelayMs: 1500,
  scrollDistance: 800,
  normalizeTargetUrl,
  getTargets,
  extractListings,
  fixtures: {
    sampleHtmlPath:   path.resolve(__dirname, '../../../data/<id>/sample.html'),
    sampleExpectedPath: path.resolve(__dirname, '../../../data/<id>/sample.expected.json'),
  },
};
```

## 4. Workflow checklist

- [ ] `src/sources/<id>/index.js` created and exported
- [ ] `npm run capture:fixture -- --source <id>` run (generates HTML + expected JSON)
- [ ] Expected JSON reviewed and corrected if needed
- [ ] `npm run validate:fixtures -- --source <id>` passes
- [ ] `npm run scrape -- --source <id>` runs against live site
- [ ] `.env.example` updated with the new env var
- [ ] One-line note added to `docs/ai-context.md` under "Registered sources"
